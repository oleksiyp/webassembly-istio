1    import {
2        proxy_get_shared_data,
3        proxy_set_shared_data
4    } from "@solo-io/proxy-runtime/imports";
5    
6    import {
7        Reference,
8        WasmResultValues,
9        log,
10       LogLevelValues
11   } from "@solo-io/proxy-runtime/runtime";
12   
13   export class RNG {
14       private static MIN: number = 0;
15       private static MAX: number = 4294967295; // u32 max
16       private initialSeed: u32;
17       private keyBuffer: ArrayBuffer;
18   
19       constructor(seed: u32, key: string) {
20           this.keyBuffer = String.UTF8.encode(key);
21           this.initialSeed = seed;
22       }
23   
24       public next(): number {
25           const seedBuf = new ArrayBuffer(4)
26           const cas = new Reference<u32>()
27           const seedU32Buf = Uint32Array.wrap(seedBuf);
28   
29           let setResult: WasmResultValues;
30           do {
31               const getResult = proxy_get_shared_data(
32                   changetype<usize>(this.keyBuffer),
33                   this.keyBuffer.byteLength,
34   
35                   changetype<usize>(seedBuf),
36                   seedBuf.byteLength,
37   
38                   cas.ptr()
39               );
40   
41               log(LogLevelValues.info, "GET RESULT: " + getResult.toString());
42   
43   
44               if (getResult !== WasmResultValues.Ok) {
45                   seedU32Buf[0] = this.initialSeed;
46                   cas.data = 0;
47               } else {
48                   seedU32Buf[0] = this.xorshift(seedU32Buf[0]);
49               }
50   
51               log(LogLevelValues.info, "SEED: " + seedU32Buf[0].toString());
52               log(LogLevelValues.info, "CAS: " + cas.data.toString());
53   
54               setResult = proxy_set_shared_data(
55                   changetype<usize>(this.keyBuffer),
56                   this.keyBuffer.byteLength,
57   
58                   changetype<usize>(seedBuf),
59                   seedBuf.byteLength,
60   
61                   cas.data
62               );
63   
64               log(LogLevelValues.info, "SET RESULT: " + setResult.toString());
65           } while (setResult !== WasmResultValues.Ok);
66   
67           log(LogLevelValues.info, "SEED: " + seedU32Buf[0].toString());
68           return (seedU32Buf[0] as number - RNG.MIN) / (RNG.MAX - RNG.MIN);
69       }
70   
71       private xorshift(value: u32): u32 {
72           // Xorshift*32
73           // Based on George Marsaglia's work: http://www.jstatsoft.org/v08/i14/paper
74           value ^= value << 13;
75           value ^= value >> 17;
76           value ^= value << 5;
77           return value;
78       }
79   }
